To check the deployment status:
  kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "zentinel-waf.name" . }}

To view logs:
  kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "zentinel-waf.name" . }} -f

To check health:
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "zentinel-waf.fullname" . }} 8080:{{ .Values.service.healthPort }}
  curl http://localhost:8080/health

To view metrics:
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "zentinel-waf.fullname" . }} 9090:{{ .Values.service.metricsPort }}
  curl http://localhost:9090/metrics
